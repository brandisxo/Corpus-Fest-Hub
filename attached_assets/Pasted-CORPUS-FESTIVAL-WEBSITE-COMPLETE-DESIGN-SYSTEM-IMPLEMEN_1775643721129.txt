SCROLL BEHAVIOR & TRANSITIONS:

Technical Implementation:
```javascript
// Scroll detection
const sections = [
  { id: 'academic', region: 'frontal', color: '#d4af78', rotation: [-5, 10, 0] },
  { id: 'sports-court', region: 'motor', color: '#7ba3d4', rotation: [0, -15, 5] },
  { id: 'sports-field', region: 'motor-wide', color: '#8bc4a0', rotation: [8, 5, -3] },
  { id: 'arts', region: 'right-hemisphere', color: '#d48b7b', rotation: [0, 20, 0] }
];

// Intersection Observer
const observer = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      const sectionData = sections[entry.target.dataset.index];
      transitionBrainState(sectionData);
    }
  });
}, { threshold: 0.6 });

function transitionBrainState(data) {
  // 1. Fade out current neural animation (300ms)
  fadeOutNeuralPaths();
  
  // 2. Rotate brain (800ms ease-out)
  rotateBrain(data.rotation);
  
  // 3. Crossfade brain color (600ms)
  morphBrainColor(data.color);
  
  // 4. Highlight new region (400ms delay, 500ms fade-in)
  setTimeout(() => {
    highlightRegion(data.region, data.color);
  }, 400);
  
  // 5. Draw new neural paths (600ms delay, 500ms draw)
  setTimeout(() => {
    drawNeuralPaths(data.region, data.color);
  }, 600);
  
  // 6. Start neuron dots (1000ms delay)
  setTimeout(() => {
    animateNeuronDots(data.color);
  }, 1000);
}
